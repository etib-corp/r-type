156 Vector3<T> refract(const Vector3<T>& n, double eta) const
157 {
158 double k = 1.0 - eta * eta * (1.0 - dot(n) * dot(n));
159 if (k < 0.0)
160 return Vector3<T>();
161 else
162 return *this * eta - n * (eta * dot(n) + sqrt(k));
163 }
