109 if (_entityToIndexMap.find(entity) == _entityToIndexMap.end()) {
110 throw ComponentManagerError("Removing non-existent component");
111 }
112
113 size_t indexOfRemovedEntity = _entityToIndexMap[entity];
114 size_t indexOfLastElement = _size - 1;
115 _components[indexOfRemovedEntity] = _components[indexOfLastElement];
116
117 Entity entityOfLastElement = _indexToEntityMap[indexOfLastElement];
118 _entityToIndexMap[entityOfLastElement] = indexOfRemovedEntity;
119 _indexToEntityMap[indexOfRemovedEntity] = entityOfLastElement;
120
121 _entityToIndexMap.erase(entity);
122 _indexToEntityMap.erase(indexOfLastElement);
123
124 _size--;
125 }
