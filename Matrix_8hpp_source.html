185 double determinant() const
186 {
187 if (WIDTH != HEIGHT) {
188 throw MatrixError("Matrix must be square to calculate determinant.");
189 }
190 if (WIDTH == 2) {
191 return this->_matrix[0][0] * this->_matrix[1][1] - this->_matrix[0][1] * this->_matrix[1][0];
192 }
193 double det = 0;
194 for (int i = 0; i < WIDTH; i++) {
195 Matrix<WIDTH - 1, HEIGHT - 1, T> submatrix;
196 for (int j = 1; j < HEIGHT; j++) {
197 for (int k = 0; k < WIDTH; k++) {
198 if (k < i) {
199 submatrix[j - 1][k] = this->_matrix[j][k];
200 } else if (k > i) {
201 submatrix[j - 1][k - 1] = this->_matrix[j][k];
202 }
203 }
204 }
205 det += this->_matrix[0][i] * submatrix.determinant() * (i % 2 == 0 ? 1 : -1);
206 }
