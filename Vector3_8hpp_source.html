186 Vector3<T> refract(const Vector3<T>& n, double eta) const
187 {
188 double k = 1.0 - eta * eta * (1.0 - dot(n) * dot(n));
189 if (k < 0.0)
190 return Vector3<T>();
191 else
192 return *this * eta - n * (eta * dot(n) + sqrt(k));
